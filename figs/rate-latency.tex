\begin{tikzpicture}[
  declare function={
    ratelatency(\R,\T,\x)= (\x<=\T) * 0   +
    (\x>\T) * (\R*(\x-\T));
    }
]
\begin{axis}[
    x=1cm,
    y=1cm,
    domain=0:5,
    ymin=0,
    xmin=0,
    axis x line=middle,
    axis y line=middle,
     xlabel style={
            anchor=west,
            at={(ticklabel* cs:1.0)},
            xshift=5pt
        },
        xlabel=$t$,
    ]
    \addplot[DodgerBlue3,ultra thick,smooth]
        {ratelatency(0.5,1,x)};
\end{axis}
\end{tikzpicture}
